\begin{frame}{Cycles}
  \begin{exampleblock}{Hamiltonian path in tournament graph \pno{3.7.18}}
    \begin{itemize}
      \item digraph $G = (V, E)$
      \item $\forall u,v: (u \to v \lor v \to u) \land \lnot (u \to v \land v \to u)$
      \item hamiltonial path
    \end{itemize}
  \end{exampleblock}

  \begin{block}{Solution.}
    \begin{itemize}
      \item existence: induction on $n$
      \item algorithm $1 + 2 + \cdots + (n-1) = O(n^{2})$
    \end{itemize}
  \end{block}
\end{frame}
