\begin{frame}
  \[
    \red{\boxed{W(n) \ge 2n - 1}}
  \]

  \pause
  \begin{columns}
    \column{0.30\textwidth}
      \centerline{\red{Adversary $\mathcal{A}$:}}
	  \[
		x > M[i][j]
	  \]
	  \[
		x = M[i][j]
	  \]
	  \[
		x < M[i][j]
	  \]
    \column{0.40\textwidth}
      \fig{width = 0.80\textwidth}{figs/adversary-alg}
    \column{0.30\textwidth}
      \centerline{\teal{Algorithm $\mathbb{A}$:}}
      \[
		\textsc{Compare}(\red{x}, M[i][j])
      \]
  \end{columns}

  \pause
  \vspace{0.50cm}
  \[
    \text{Diagonals: } i + j = n - 1 \quad\&\quad i + j = n
  \]

  \pause
  \begin{center}
	{\purple{No particular ordering requirements on these two diagonals!}}
  \end{center}

  \pause
  \vspace{-0.30cm}
  \begin{align*}
    i + j \le n - 1 &\implies x > M_{ij} \\
    i + j > n - 1 &\implies x < M_{ij} \\
  \end{align*}
\end{frame}
