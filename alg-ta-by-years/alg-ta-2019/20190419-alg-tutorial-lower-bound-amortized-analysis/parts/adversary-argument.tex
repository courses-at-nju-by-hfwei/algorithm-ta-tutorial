\begin{frame}{}
  \[
    \text{Assume } M: n \times n 
  \]

  \[
    W(n) \le 2n - 1
  \]

  \pause
  \[
    \boxed{\red{W(n) \ge 2n - 1}}
  \]

  \pause
  \centerline{\teal{By Adversary Argument!}}

  \pause
  \[
    \text{Diagonals: } i + j = n - 1 \quad\&\quad i + j = n
  \]

  \pause
  \centerline{\teal{No particular ordering requirements on these two diagonals!}}

  \pause
  \begin{align*}
    i + j \le n - 1 &\implies x > M_{ij} \\
    i + j > n - 1 &\implies x < M_{ij} \\
  \end{align*}
\end{frame}
